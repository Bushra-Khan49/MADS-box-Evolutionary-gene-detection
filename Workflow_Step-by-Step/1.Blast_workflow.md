# BLAST Workflow — Step 01 (Arabidopsis → Rice example)

**Purpose**  
Make a BLAST DB from a species proteome (example: rice), query Arabidopsis ABCDE genes, remove redundancies and visualize mapping statistics.

**Prereqs**
- BLAST+ installed (`makeblastdb`, `blastp`, `blastdbcmd`)
- `cd-hit` (for redundancy removal) or `seqkit`
- `awk`, `grep`, `sort`, `uniq`
- R with `data.table`, `ggplot2`
- Optional: `seqkit` for fasta extraction

## 1) Create directories
```bash
# from your repo root
mkdir -p workflow-step-by-step/{data,db,input,results}


## 2) Build BLAST DB for rice
makeblastdb -in workflow-step-by-step/data/Oryza_sativa_proteome.fa \
            -dbtype prot \
            -parse_seqids \
            -out workflow-step-by-step/db/rice_db


## 3) BLAST: Arabidopsis -> Rice
blastp -query workflow-step-by-step/input/Arabidopsis_ABCDE.fa \
       -db workflow-step-by-step/db/rice_db \
       -out workflow-step-by-step/results/arab_vs_rice.outfmt7 \
       -evalue 1e-5 \
       -outfmt 7 \
       -num_threads 4
## 4) Extract subject IDs (unique rice hits)
# get subject (sseqid) column (outfmt 7 -> tabular lines; col2 is subject)
awk '!/^#/ {print $2}' workflow-step-by-step/results/arab_vs_rice.outfmt7 \
  | sort | uniq > workflow-step-by-step/results/rice_candidate_ids.txt

## 5) Retrieve candidate sequences from rice DB
blastdbcmd -db workflow-step-by-step/db/rice_db \
           -entry_batch workflow-step-by-step/results/rice_candidate_ids.txt \
           -out workflow-step-by-step/results/rice_candidate_seqs.fa
                                **OR**
seqkit grep -f workflow-step-by-step/results/rice_candidate_ids.txt workflow-step-by-step/data/Oryza_sativa_proteome.fa \
  -o workflow-step-by-step/results/rice_candidate_seqs.fa

## 6) Remove redundant sequences (CD-HIT clustering, 95% identity)
cd-hit -i workflow-step-by-step/results/rice_candidate_seqs.fa \
       -o workflow-step-by-step/results/rice_candidates_cdhit95.fa \
       -c 0.95 -n 5 -d 0 -T 4 -M 0
# -M 0 uses all memory; adjust -T threads as needed

## 7) (Optional) Self-blast dedupe check
# create a DB of rice candidates
makeblastdb -in workflow-step-by-step/results/rice_candidate_seqs.fa -dbtype prot -out workflow-step-by-step/db/rice_candidates_db

blastp -query workflow-step-by-step/results/rice_candidate_seqs.fa \
       -db workflow-step-by-step/db/rice_candidates_db \
       -out workflow-step-by-step/results/rice_candidates_selfblast.outfmt7 \
       -evalue 1e-5 -outfmt 7 -num_threads 4

## 8) Move final non-redundant sequences to results and record logs

## 9) Visualization (R)
# plot_blast_summary.R
library(data.table); library(ggplot2)

# helper to read outfmt7 robustly:
read_outfmt7 <- function(path) {
  lines <- readLines(path)
  lines <- lines[!grepl("^#", lines)]
  if (length(lines)==0) return(data.table())
  dt <- fread(text = paste(lines, collapse = "\n"), sep="\t", header=FALSE)
  setnames(dt, c("qseqid","sseqid","pident","length","mismatch","gapopen",
                 "qstart","qend","sstart","send","evalue","bitscore"))
  return(dt)
}

# If you have one file (arab_vs_rice.outfmt7):
blast <- read_outfmt7("workflow-step-by-step/results/arab_vs_rice.outfmt7")
if (nrow(blast)==0) stop("No BLAST hits found - check file.")

# Add species column — here single species 'Oryza_sativa'
blast[, species := "Oryza_sativa"]

# Graph 1: scatter / bubble plot: bitscore vs %identity colored by -log10(evalue)
blast[, neglog10_e := -log10(pmax(evalue, 1e-200))]  # avoid log(0)

p1 <- ggplot(blast, aes(x=bitscore, y=pident, color = neglog10_e, size = bitscore)) +
  geom_point(alpha=0.7) +
  theme_minimal() +
  labs(title="Arabidopsis ABCDE → Rice hits",
       x="Bit score", y="% Identity", color="-log10(E-value)", size="Bit score")
ggsave("workflow-step-by-step/results/plots/arab_vs_rice_scatter.png", p1, width=8, height=6)

# Graph 2: ratio of rice hits per Arabidopsis gene (or ratio wrt total proteome)
# Option A: hits per Arabidopsis query
hits_per_query <- blast[, .N, by=qseqid]
p2 <- ggplot(hits_per_query, aes(x=qseqid, y=N)) +
  geom_col(fill="steelblue") +
  theme_minimal() +
  labs(title="Number of Rice hits per Arabidopsis ABCDE gene",
       x="Arabidopsis query gene", y="Count of Rice hits") +
  theme(axis.text.x = element_text(angle=45, hjust=1))
dir.create("workflow-step-by-step/results/plots", showWarnings = FALSE, recursive = TRUE)
ggsave("workflow-step-by-step/results/plots/hits_per_query_barplot.png", p2, width=8, height=5)

# Option B: compute ratio = number_of_mapped / total_protein_count (requires totals file)
# If you have totals file (tab: species<TAB>total_proteins), load and compute
totals_file <- "workflow-step-by-step/results/totals.tsv"
if (file.exists(totals_file)) {
  totals <- fread(totals_file, header=FALSE)
  setnames(totals, c("species","total_proteins"))
  mapped <- blast[, .(mapped = uniqueN(sseqid)), by=species]
  merged <- merge(mapped, totals, by="species", all.x=TRUE)
  merged[, ratio := mapped / total_proteins]
  p3 <- ggplot(merged, aes(x=species, y=ratio)) +
    geom_col(fill="darkgreen") +
    theme_minimal() +
    labs(title="Ratio of mapped proteins to total proteins", x="Species", y="Mapped/Total")
  ggsave("workflow-step-by-step/results/plots/mapped_ratio_barplot.png", p3, width=6, height=4)
}
