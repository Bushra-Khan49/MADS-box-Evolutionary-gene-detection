### Procedure


```bash
cd ~/Desktop/NIPGR-data/BLAST_project

# Create a BLAST database for Arabidopsis proteome
makeblastdb -in Arabidopsis_thaliana.fa -dbtype prot -out at_db

# Perform Blastp search against the Arabidopsis reference
blastp -query At_ABCDE_QUERY.fa -db at_db -out Arabidopsis_Blast_results.txt -evalue 1e-5 -outfmt 7
```

# Visualisation using pandas and matplotlib in Python

```python
[![Open in Colab](https://colab.research.google.com/assets/colab-badge.svg)](https://colab.research.google.com/drive/1CfAzNoKbsFE9PxtaFEIyFNVFZ4u5ZGUZ#scrollTo=cG2hhQg035Fd&line=1&uniqifier=1)


# Step 1: Import libraries
import pandas as pd
import matplotlib.pyplot as plt

# Step 2: Load file
df = pd.read_csv("Arabidopsis_Blast_results.csv")

# Step 3: Process columns
df['Query_Gene'] = df['Query ID'].astype(str).str.split().str[-1]
df['Subject_Gene'] = df['Subject ID'].astype(str)

df = df.dropna(subset=['Query_Gene', 'Subject_Gene'])
df = df[(df['Query_Gene'] != "nan") & (df['Subject_Gene'] != "nan")]

df['Bit Score'] = pd.to_numeric(df['Bit Score'], errors='coerce')
df['% Identity'] = pd.to_numeric(df['% Identity'], errors='coerce')
df['E-value'] = pd.to_numeric(df['E-value'], errors='coerce')
df = df.dropna(subset=['Bit Score', '% Identity', 'E-value'])

# Step 4: Map gene names to numeric positions for plotting
x_labels = df['Query_Gene'].unique()
y_labels = df['Subject_Gene'].unique()

x_map = {label: i for i, label in enumerate(x_labels)}
y_map = {label: i for i, label in enumerate(y_labels)}

plt.figure(figsize=(14,10))

# Step 5: Plot scatter with mapped positions
scatter = plt.scatter(
    df['Query_Gene'].map(x_map),
    df['Subject_Gene'].map(y_map),
    s=df['Bit Score']/2,
    c=df['% Identity'],
    cmap="viridis", alpha=0.7, edgecolors="k"
)

plt.colorbar(scatter, label="% Identity")
plt.xticks(list(x_map.values()), list(x_map.keys()), rotation=45, ha='right')
plt.yticks(list(y_map.values()), list(y_map.keys()))
plt.xlabel("Arabidopsis Genes (Query)")
plt.ylabel("Arabidopsis Genes (Subject)")
plt.title("Arabidopsis Self-BLAST Comparison")

# Step 6: Annotate with E-value
for i, row in df.iterrows():
    plt.text(
        x_map[row['Query_Gene']] - 0.3,
        y_map[row['Subject_Gene']],
        f"{row['E-value']:.1e}",
        fontsize=6, ha='right', va='center', color="black"
    )

plt.show()
```
